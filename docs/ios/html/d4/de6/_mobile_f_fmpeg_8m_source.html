87  strcpy(commandCharPArray[0], [LIB_NAME UTF8String]);
88 
89  for (int i=0; i < [arguments count]; i++) {
90  NSString *argument = [arguments objectAtIndex:i];

93 
94  // RUN
95  lastReturnCode = execute(([arguments count] + 1), commandCharPArray);
96 
97  // CLEANUP
