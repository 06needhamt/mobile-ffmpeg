233 void callbackBlockFunction() {
234  NSLog(@"Async callback block started.\n");
235 
236  while(redirectionEnabled) {
237  @try {
238 
239  CallbackData *callbackData = callbackDataRemove();
240  if (callbackData != nil) {
241 
242  if ([callbackData getType] == LogType) {
243 
244  // LOG CALLBACK
245  int activeLogLevel = [MobileFFmpegConfig getLogLevel];
246 
247  if (activeLogLevel == AV_LOG_QUIET || [callbackData getLogLevel] > activeLogLevel) {
248 
249  // LOG NEITHER PRINTED NOR FORWARDED
250  } else {
251 
252  if (logDelegate != nil) {
253 
254  // FORWARD LOG TO DELEGATE
255  [logDelegate logCallback:[callbackData getLogLevel]:[callbackData getLogData]];
256 
257  } else {
258 
259  // WRITE TO NSLOG
260  NSLog(@"%@: %@", [MobileFFmpegConfig logLevelToString:[callbackData getLogLevel]], [callbackData getLogData]);
261  }
262 
263  }
264 
265  } else {
266 
267  // STATISTICS CALLBACK
268  Statistics *newStatistics = [[Statistics alloc] initWithVideoFrameNumber:[callbackData getStatisticsFrameNumber] fps:[callbackData getStatisticsFps] quality:[callbackData getStatisticsQuality] size:[callbackData getStatisticsSize] time:[callbackData getStatisticsTime] bitrate:[callbackData getStatisticsBitrate] speed:[callbackData getStatisticsSpeed]];
269  [lastReceivedStatistics update:newStatistics];
270 
271  if (logDelegate != nil) {
272 
273  // FORWARD STATISTICS TO DELEGATE
274  [statisticsDelegate statisticsCallback:lastReceivedStatistics];
275  }
276  }
277 
278  } else {
279  callbackWait(100);
280  }
281 
282  } @catch(NSException *exception) {
283  NSLog(@"Async callback block received error: %@n\n", exception);
284  NSLog(@"%@", [exception callStackSymbols]);
285  }
286  }
287 
288  NSLog(@"Async callback block stopped.\n");
289 }

441 + (void)setFontDirectory: (NSString*)fontDirectoryPath with:(NSDictionary*)fontNameMapping {
442  NSError *error = nil;
443  BOOL isDirectory = YES;
444  BOOL isFile = NO;
445  int validFontNameMappingCount = 0;
446  NSString *tempConfigurationDirectory = [NSTemporaryDirectory() stringByAppendingPathComponent:@".mobileffmpeg"];
447  NSString *fontConfigurationFile = [tempConfigurationDirectory stringByAppendingPathComponent:@"fonts.conf"];
448 
449  if (![[NSFileManager defaultManager] fileExistsAtPath:tempConfigurationDirectory isDirectory:&isDirectory]) {
450 
451  if (![[NSFileManager defaultManager] createDirectoryAtPath:tempConfigurationDirectory withIntermediateDirectories:YES attributes:nil error:&error]) {
452  NSLog(@"Failed to set font directory. Error received while creating temp conf directory: %@.", error);
453  return;
454  }
455 
456  NSLog(@"Created temporary font conf directory: TRUE.");
457  }
458 
459  if ([[NSFileManager defaultManager] fileExistsAtPath:fontConfigurationFile isDirectory:&isFile]) {
460  BOOL fontConfigurationDeleted = [[NSFileManager defaultManager] removeItemAtPath:fontConfigurationFile error:NULL];
461  NSLog(@"Deleted old temporary font configuration: %s.", fontConfigurationDeleted?"TRUE":"FALSE");
462  }
463 
464  /* PROCESS MAPPINGS FIRST */
465  NSString *fontNameMappingBlock = @"";
466  for (NSString *fontName in [fontNameMapping allKeys]) {
467  NSString *mappedFontName = [fontNameMapping objectForKey:fontName];
468 
469  if ((fontName != nil) && (mappedFontName != nil) && ([fontName length] > 0) && ([mappedFontName length] > 0)) {
470 
471  fontNameMappingBlock = [NSString stringWithFormat:@"%@\n%@\n%@%@%@\n%@\n%@\n%@%@%@\n%@\n%@\n",
472  @" <match target=\"pattern\">",
473  @" <test qual=\"any\" name=\"family\">",
474  @" <string>", fontName, @"</string>",
475  @" </test>",
476  @" <edit name=\"family\" mode=\"assign\" binding=\"same\">",
477  @" <string>", mappedFontName, @"</string>",
478  @" </edit>",
479  @" </match>"];
480 
481  validFontNameMappingCount++;
482  }
483  }
484 
485  NSString *fontConfiguration = [NSString stringWithFormat:@"%@\n%@\n%@\n%@\n%@%@%@\n%@\n%@\n",
486  @"<?xml version=\"1.0\"?>",
487  @"<!DOCTYPE fontconfig SYSTEM \"fonts.dtd\">",
488  @"<fontconfig>",
489  @" <dir>.</dir>",
490  @" <dir>", fontDirectoryPath, @"</dir>",

492  @"</fontconfig>"];
493

495  NSLog(@"Failed to set font directory. Error received while saving font configuration: %@.", error);
496  return;
