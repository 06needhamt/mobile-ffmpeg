239 void callbackBlockFunction() {
240  NSLog(@"Async callback block started.\n");
241 
242  while(redirectionEnabled) {
243  @try {
244 
245  CallbackData *callbackData = callbackDataRemove();
246  if (callbackData != nil) {
247 
248  if ([callbackData getType] == LogType) {
249 
250  // LOG CALLBACK
251  int activeLogLevel = [MobileFFmpegConfig getLogLevel];
252 
253  if (runningSystemCommand == 1) {
254 
255  // REDIRECT SYSTEM OUTPUT
256  if ((activeLogLevel != AV_LOG_QUIET) && ([callbackData getLogLevel] <= activeLogLevel)) {
257  [systemCommandOutput appendString:[callbackData getLogData]];
258  }
259  } else if ((activeLogLevel == AV_LOG_QUIET) || ([callbackData getLogLevel] > activeLogLevel)) {
260 
261  // LOG NEITHER PRINTED NOR FORWARDED
262  } else {
263 
264  // ALWAYS REDIRECT COMMAND OUTPUT
265  [lastCommandOutput appendString:[callbackData getLogData]];
266 
267  if (logDelegate != nil) {
268 
269  // FORWARD LOG TO DELEGATE
270  [logDelegate logCallback:[callbackData getLogLevel]:[callbackData getLogData]];
271 
272  } else {
273 
274  // WRITE TO NSLOG
275  NSLog(@"%@: %@", [MobileFFmpegConfig logLevelToString:[callbackData getLogLevel]], [callbackData getLogData]);
276  }
277 
278  }
279 
280  } else {
281 
282  // STATISTICS CALLBACK
283  Statistics *newStatistics = [[Statistics alloc] initWithVideoFrameNumber:[callbackData getStatisticsFrameNumber] fps:[callbackData getStatisticsFps] quality:[callbackData getStatisticsQuality] size:[callbackData getStatisticsSize] time:[callbackData getStatisticsTime] bitrate:[callbackData getStatisticsBitrate] speed:[callbackData getStatisticsSpeed]];
284  [lastReceivedStatistics update:newStatistics];
285 
286  if (logDelegate != nil) {
287 
288  // FORWARD STATISTICS TO DELEGATE
289  [statisticsDelegate statisticsCallback:lastReceivedStatistics];
290  }
291  }
292 
293  } else {
294  callbackWait(100);
295  }
296 
297  } @catch(NSException *exception) {
298  NSLog(@"Async callback block received error: %@n\n", exception);
299  NSLog(@"%@", [exception callStackSymbols]);
300  }
301  }
302 
303  NSLog(@"Async callback block stopped.\n");
304 }

325 int mobileffmpeg_system_execute(NSArray *arguments, NSArray *commandOutputEndPatternList, long timeout) {
326  systemCommandOutput = [[NSMutableString alloc] init];
327  runningSystemCommand = 1;
328 
329  int rc = [MobileFFmpeg executeWithArguments:arguments];
330 
331  long totalWaitTime = 0;
332 
333  while ((systemCommandOutputContainsPattern(commandOutputEndPatternList) == 0) && (totalWaitTime < timeout)) {
334  [NSThread sleepForTimeInterval:.02];
335  totalWaitTime += 20;
336  }
337 
338  runningSystemCommand = 0;
339 
340  [MobileFFmpeg cancel];
341 
342  return rc;

542 + (void)setFontDirectory: (NSString*)fontDirectoryPath with:(NSDictionary*)fontNameMapping {
543  NSError *error = nil;
544  BOOL isDirectory = YES;
545  BOOL isFile = NO;
546  int validFontNameMappingCount = 0;
547  NSString *tempConfigurationDirectory = [NSTemporaryDirectory() stringByAppendingPathComponent:@".mobileffmpeg"];
548  NSString *fontConfigurationFile = [tempConfigurationDirectory stringByAppendingPathComponent:@"fonts.conf"];
549 
550  if (![[NSFileManager defaultManager] fileExistsAtPath:tempConfigurationDirectory isDirectory:&isDirectory]) {
551 
552  if (![[NSFileManager defaultManager] createDirectoryAtPath:tempConfigurationDirectory withIntermediateDirectories:YES attributes:nil error:&error]) {
553  NSLog(@"Failed to set font directory. Error received while creating temp conf directory: %@.", error);
554  return;
555  }
556 
557  NSLog(@"Created temporary font conf directory: TRUE.");
558  }
559 
560  if ([[NSFileManager defaultManager] fileExistsAtPath:fontConfigurationFile isDirectory:&isFile]) {
561  BOOL fontConfigurationDeleted = [[NSFileManager defaultManager] removeItemAtPath:fontConfigurationFile error:NULL];
562  NSLog(@"Deleted old temporary font configuration: %s.", fontConfigurationDeleted?"TRUE":"FALSE");
563  }
564 
565  /* PROCESS MAPPINGS FIRST */
566  NSString *fontNameMappingBlock = @"";
567  for (NSString *fontName in [fontNameMapping allKeys]) {
568  NSString *mappedFontName = [fontNameMapping objectForKey:fontName];
569 
570  if ((fontName != nil) && (mappedFontName != nil) && ([fontName length] > 0) && ([mappedFontName length] > 0)) {
571 
572  fontNameMappingBlock = [NSString stringWithFormat:@"%@\n%@\n%@%@%@\n%@\n%@\n%@%@%@\n%@\n%@\n",
573  @" <match target=\"pattern\">",
574  @" <test qual=\"any\" name=\"family\">",
575  @" <string>", fontName, @"</string>",
576  @" </test>",
577  @" <edit name=\"family\" mode=\"assign\" binding=\"same\">",
578  @" <string>", mappedFontName, @"</string>",
579  @" </edit>",
580  @" </match>"];
581 
582  validFontNameMappingCount++;
583  }
584  }
585 
586  NSString *fontConfiguration = [NSString stringWithFormat:@"%@\n%@\n%@\n%@\n%@%@%@\n%@\n%@\n",
587  @"<?xml version=\"1.0\"?>",
588  @"<!DOCTYPE fontconfig SYSTEM \"fonts.dtd\">",
589  @"<fontconfig>",
590  @" <dir>.</dir>",
591  @" <dir>", fontDirectoryPath, @"</dir>",
592  fontNameMappingBlock,
593  @"</fontconfig>"];
594 
595  if (![fontConfiguration writeToFile:fontConfigurationFile atomically:YES encoding:NSUTF8StringEncoding error:&error]) {
596  NSLog(@"Failed to set font directory. Error received while saving font configuration: %@.", error);
597  return;
598  }
599  NSLog(@"Saved new temporary font configuration with %d font name mappings.", validFontNameMappingCount);
600 
601  [MobileFFmpegConfig setFontconfigConfigurationPath:tempConfigurationDirectory];
602 
603  NSLog(@"Font directory %@ registered successfully.", fontDirectoryPath);
